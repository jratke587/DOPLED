210 static size_t rmt_encode_dopled(
211 rmt_encoder_t *encoder,
212 rmt_channel_handle_t channel,
213 const void *primary_data,
214 size_t data_size,
215 rmt_encode_state_t *ret_state)
216 {
217 dopled_encoder_t *led_encoder = __containerof(encoder, dopled_encoder_t, base);
218 rmt_encoder_handle_t bytes_encoder = led_encoder->bytes_encoder;
219 rmt_encoder_handle_t copy_encoder = led_encoder->copy_encoder;
220 rmt_encode_state_t session_state = RMT_ENCODING_RESET;
221 rmt_encode_state_t state = RMT_ENCODING_RESET;
222 size_t encoded_symbols = 0;
223 switch (led_encoder->state)
224 {
225 case 0: // leading symbol
226 encoded_symbols += copy_encoder->encode(
227 copy_encoder,
228 channel,
229 &led_encoder->leading_symbol,
230 sizeof(rmt_symbol_word_t),
231 &session_state);
232 if (session_state & RMT_ENCODING_COMPLETE)
233 {
234 led_encoder->state = 1; // we can only switch to the next state when the current encoder finished
235 }
236 if (session_state & RMT_ENCODING_MEM_FULL)
237 {
238 state = RMT_ENCODING_MEM_FULL;
239 goto out; // yield if there is no free space to put other encoding artifacts
240 }
241 // fall-through
242 case 1: // send data
243 encoded_symbols += bytes_encoder->encode(bytes_encoder, channel, primary_data, data_size, &session_state);
244 if (session_state & RMT_ENCODING_COMPLETE)
245 {
246 led_encoder->state = 2; // switch to next state when current encoding session finished
247 }
248 if (session_state & RMT_ENCODING_MEM_FULL)
249 {
250 state = RMT_ENCODING_MEM_FULL;
251 goto out; // yield if there is no free space to put other encoding artifacts
252 }
253 // fall-through
254 case 2: // send reset code
255 encoded_symbols += copy_encoder->encode(
256 copy_encoder,
257 channel,
258 &led_encoder->ending_symbol,
259 sizeof(rmt_symbol_word_t),
260 &session_state);
261 if (session_state & RMT_ENCODING_COMPLETE)
262 {
263 led_encoder->state = RMT_ENCODING_RESET; // back to the initial encoding session
264 state = RMT_ENCODING_COMPLETE; // telling the caller the encoding has finished
265 }
266 if (session_state & RMT_ENCODING_MEM_FULL)
267 {
268 state = RMT_ENCODING_MEM_FULL;
269 goto out; // yield if there is no free space to put other encoding artifacts
270 }
271 }
272 out:
273 *ret_state = state;
274 return encoded_symbols;
275 }
