182 static size_t rmt_encode_dopled(

184 rmt_channel_handle_t channel,
185 const void *primary_data,
186 size_t data_size,
187 rmt_encode_state_t *ret_state)
188 {
189 dopled_encoder_t *led_encoder = __containerof(encoder, dopled_encoder_t, base);
190 rmt_encoder_handle_t bytes_encoder = led_encoder->bytes_encoder;
191 rmt_encoder_handle_t copy_encoder = led_encoder->copy_encoder;
192 rmt_encode_state_t session_state = RMT_ENCODING_RESET;
193 rmt_encode_state_t state = RMT_ENCODING_RESET;
194 size_t encoded_symbols = 0;
195 switch (led_encoder->state)
196 {
197 case 0: // leading symbol
198 encoded_symbols += copy_encoder->encode(
199 copy_encoder,
200 channel,
201 &led_encoder->leading_symbol,
202 sizeof(rmt_symbol_word_t),
203 &session_state);
204 if (session_state & RMT_ENCODING_COMPLETE)
205 {
206 led_encoder->state = 1; // we can only switch to the next state when the current encoder finished
207 }
208 if (session_state & RMT_ENCODING_MEM_FULL)
209 {
210 state = RMT_ENCODING_MEM_FULL;
211 goto out; // yield if there is no free space to put other encoding artifacts
212 }
213 // fall-through
214 case 1: // send data
215 encoded_symbols += bytes_encoder->encode(bytes_encoder, channel, primary_data, data_size, &session_state);
216 if (session_state & RMT_ENCODING_COMPLETE)
217 {
218 led_encoder->state = 2; // switch to next state when current encoding session finished
219 }
220 if (session_state & RMT_ENCODING_MEM_FULL)
221 {
222 state = RMT_ENCODING_MEM_FULL;
223 goto out; // yield if there is no free space to put other encoding artifacts
224 }
225 // fall-through
226 case 2: // send reset code
227 encoded_symbols += copy_encoder->encode(
228 copy_encoder,
229 channel,
230 &led_encoder->ending_symbol,
231 sizeof(rmt_symbol_word_t),
232 &session_state);
233 if (session_state & RMT_ENCODING_COMPLETE)
234 {
235 led_encoder->state = RMT_ENCODING_RESET; // back to the initial encoding session
236 state = RMT_ENCODING_COMPLETE; // telling the caller the encoding has finished
237 }
238 if (session_state & RMT_ENCODING_MEM_FULL)
239 {
240 state = RMT_ENCODING_MEM_FULL;
241 goto out; // yield if there is no free space to put other encoding artifacts
242 }
243 }
